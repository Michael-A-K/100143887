import math
def periRect(l, w):
    return l*2 + w*2

def trapArea(b1, b2, h):
    return (0.5*(b1+b2))*h

def conicVol(r, h):
    return (math.pi*(r*r)*(h/3))

def cubicBarrier(s):
    return s*s*s

def blastArea(r):
    return math.pi *(r*r)

def elevationVect(x1, y1, x2, y2):
    return (y2-y1)/(x2-x1)

def atmoShift(f):
    return (f-32)*(5/9)
